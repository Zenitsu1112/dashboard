async function initDb() { try { const manifestResponse = await fetch('Secret/manifest.json'); const files = await manifestResponse.json(); for (const file of files) { const res = await fetch(file); let rawData = await res.json(); // --- SECTOR 1: CHAWKIDAR (The Double-Column Splitter) --- if (file.toLowerCase().includes('chawkidar')) { rawData.forEach(row => { if (row["__EMPTY_2"] && row["__EMPTY_2"] !== "Naw•") { localDb.push(sanitizeEntry({ "Name": row["__EMPTY_2"], "Father": row["__EMPTY_3"], "Village": "Sahibganj (Sector A)", "Role": "Village Security (Chawkidar)", "ID": row["__EMPTY_1"] })); } if (row["__EMPTY_6"] || row["__EMPTY_7"]) { localDb.push(sanitizeEntry({ "Name": row["__EMPTY_6"] || "UNKNOWN", "Father": row["__EMPTY_7"], "Village": "Sahibganj (Sector B)", "Role": "Village Security (Chawkidar)", "ID": row["__EMPTY_5"] })); } }); } // --- SECTOR 2: SEED (The Structured List) --- else if (file.toLowerCase().includes('seed')) { rawData.forEach(row => { if (row["__EMPTY_4"]) { localDb.push(sanitizeEntry({ "Name": row["__EMPTY_4"], "Father": row["__EMPTY_5"], "Village": row["__EMPTY_7"] || row["__EMPTY_9"], "Role": "Agriculture (Beneficiary)", "ID": row["__EMPTY_6"] // Phone number as ID here })); } }); } // --- SECTOR 3: TAX DATABASE (The Original Registry) --- else { rawData.forEach(row => { localDb.push({ "Name": row["Owner Name"] || row["Name"], "Father": row["Father Name"] || row["Father"], "Village": row["Village"] || "N/A", "Role": row["Vehicle Category"] || "Tax Registry", "ID": row["Registration No"] || "N/A" }); }); } } printToConsole(`SUCCESS: ${localDb.length} IDENTITIES INDEXED.`); } catch (e) { printToConsole("CRITICAL_ERROR: MANIFEST_UNREADABLE."); } } // --- THE AUTO-CORRECTOR (Cleaning OCR Glitches) --- function sanitizeEntry(obj) { const corrections = { "TUDf/": "TUDU", "TUDV": "TUDU", "SORER": "SOREN", "SOREO": "SOREN", "HEMBROM": "HEMBRAM", "MARAI4DI": "MARANDI", "KISKV": "KISKU" }; for (let field in obj) { if (typeof obj[field] === 'string') { let text = obj[field].toUpperCase(); for (let wrong in corrections) { text = text.split(wrong).join(corrections[wrong]); } obj[field] = text.trim(); } } return obj; }
